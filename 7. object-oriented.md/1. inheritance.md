> Note: Multiple inheritance is not supported in C# through class.

        using System;  
        namespace classobjects  
        {         
            class Animal{
                public String name;
                public void eat(){
                    Console.WriteLine($"{this.name} eats");
                }
            }
            class Dog : Animal{

            }
            class Test  {                                                           
                static void Main(string[] args){        
                    Dog d = new Dog();
                    d.name = "tom";
                    d.eat();
                }     
            }  
        }          
        
        output : tom eats
